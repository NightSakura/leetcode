1、两数之和：考察map的基本用法
```go
func twoSum(nums []int, target int) []int {
   var res []int
   numMap := make(map[int] int)
   for cur,val := range nums {
      need := target - val
      pre,flag := numMap[need]
      if flag {
         res = []int{pre,cur}
         break
      }else{
         numMap[val] = cur
      }
   }
   return res
}
```
2、两数相加：链表的使用方法，注意处理好溢出和循环的终止条件
```go
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
   head := &ListNode{0,nil}
   cur,overflow := head,0

   for l1 != nil || l2 != nil || overflow !=0 {
      n1,n2 := 0,0
      if l1 != nil {
         n1,l1 = l1.Val, l1.Next
      }
      if l2 != nil {
         n2,l2 = l2.Val, l2.Next
      }

      if (n1+n2+overflow) >= 10 {
         cur.Next = &ListNode{n1+n2+overflow-10,nil}
         overflow = 1
      }else{
         cur.Next = &ListNode{n1+n2+overflow,nil}
         overflow = 0
      }
      cur = cur.Next
   }
   return head.Next
}
```

3、最长不重复子串的长度，记录当前串的开始位置，注意重复条件不要取错，另外遍历完串要计算重点减去开始位置的长度，然后返回最大长度
```go
func lengthOfLongestSubstring(s string) int {
	cout,start := 0,0
	numMap := make(map[rune] int)
	for item,val := range s {
		index,ok := numMap[val]
		if ok && index >= start {
			length := item - start
			if length > cout {
				cout = length
			}
			start = index + 1

		}
		numMap[val] = item
	}
	if len(s) - start > cout {
		cout = len(s) - start
	}

	return cout
}
```
