
21、合并有序链表，按顺序取然后合并就行
```go
func mergeTwoLists(l1 *ListNode, l2 *ListNode) *ListNode {
	const INT_MAX = int(^uint(0) >> 1)
	head := &ListNode{0,nil}
	cur := head
	for l1 != nil || l2 != nil {
		n1,n2 := INT_MAX,INT_MAX
		if l1 != nil {
			n1 = l1.Val
		}
		if l2 != nil {
			n2 = l2.Val
		}
		if n1 < n2 {
			cur.Next = l1
			l1 = l1.Next
		} else {
			cur.Next = l2
			l2 = l2.Next
		}
		cur = cur.Next
	}
	return head.Next
}
```
